package junit;
import java.util.Random;
import java.util.Scanner;

public class Assignment {
    Random random = new Random();
    public void input(){
        String splitNumber = null ;
        Scanner scan = new Scanner(System.in);
        System.out.println("Enter a String :");
        String word = scan.next();
        int wordLength = word.length();
        if(wordLength > 2){
            splitNumber = word.substring(0,2);
            System.out.println(splitNumber);
        }else{
            System.out.println(word);
        }
        
        System.out.println();
        
        System.out.println("How many times do you want to print the random numbers :");
        int number = scan.nextInt();
        int randomValue;
        for(int i = 0; i < number; i++){
            randomValue = random.nextInt((999 - 100) + 1) + 100;
            System.out.println(splitNumber + ": " + randomValue);
        }
        
    }
    
    public static void main(String[] args) {
        Assignment obj = new Assignment();
        obj.input();
    }
}
